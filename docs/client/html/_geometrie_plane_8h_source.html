117 static const pair<double, double>
118 resoudre_equation_second_degre(const double &a, const double &b,
119 const double &c) {
120 double delta = b * b - 4 * a * c;
121 if (delta <= 0)
122 throw ExceptionGeometriePlane(ExceptionGeometriePlane::DELTA_NEGATIF);
123 double sqrt_delta = sqrt(delta);
124 auto x1 = (-b - sqrt_delta) / (2 * a);
125 auto x2 = (-b + sqrt_delta) / (2 * a);
